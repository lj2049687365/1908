#include<stdio.h>
int a[1000][1000];
int main()
{
	int i,j,m,n,t;
	printf("请输入行数：\n");
	scanf("%d",&m);
	printf("请输入列数：\n");
	scanf("%d",&n);
	printf("请输入矩阵元素（按行列顺序）：\n"); 
	for(i=0;i<m;i++)
	for(j=0;j<n;j++)
	{
		scanf("%d",&a[i][j]);
	}
	printf("您输入的矩阵为：\n"); 
	for(i=0;i<m;i++)
	{
		for(j=0;j<n;j++)
		{
			printf("%d\t",a[i][j]);
		}
		printf("\n");
	}
	for(i=0;i<m;i++)
	for(j=0;j<n;j++)
	{
		t=a[j][i];
		a[j][i]=a[i][j];
		a[i][j]=t;
	}
	printf("转置矩阵为：\n");
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			printf("%d\t",a[i][j]);
		}
		printf("\n");
	}
}
